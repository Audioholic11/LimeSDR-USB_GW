00058 alt\_putcharbuf(\textcolor{keywordtype}{int} c)
00059 \{
00060     buf[fill\_index++] = (char)(c & 0xff);
00061     \textcolor{keywordflow}{if}(fill\_index >= ALT\_PUTBUF\_SIZE)
00062         alt\_putbufflush();
